function setproxy() {
  host_ip=$(cat /etc/resolv.conf |grep "nameserver" | awk '{print $2}')
  export http_proxy=http://$host_ip:7890
  export https_proxy=http://$host_ip:7890
  echo "Proxy set up"
  echo $(env | grep -i proxy)
}

function unsetproxy() {
  unset http_proxy
  unset https_proxy

  echo "Proxy canceled"
}
