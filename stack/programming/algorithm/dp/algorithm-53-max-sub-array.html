        int max_sub = INT32_MIN;
        for(int i = 0; i < nums.size(); ++i) {
            int sub = 0;

class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int n = nums.size();
        vector<int> dp(n);
        dp[0] = nums[0];
        int max_sub = dp[0];
        
        for(int i = 1; i < n; ++i) {
            dp[i] = max(dp[i-1] + nums[i], nums[i]);
            if(dp[i] > max_sub) {
                max_sub = dp[i];
            }
        }

        return max_sub;
    }
};

class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int dp = 0;
        int max_sub = nums[0];
        
        for(const auto i : nums) {
            dp = max(dp + i, i);
            if(dp > max_sub) {
                max_sub = dp;
            }
        }

        return max_sub;
    }
};
